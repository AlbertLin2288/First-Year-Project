    \subsection{Some definitions}

    % Note:

    \begin{definition}
        We call \(m\)-many subspaces \(\vsS_1, \vsS_2, \dots, \vsS_m \subset \vsV\) independent if and only if the equation 
        \[0 = \vec{s}_1 + \vec{s}_2 + \cdots + \vec{s} _m , \ \vec{s}_i \in \vsS_i \] 
        has only the solution \( \vec{s}_1, \vec{s}_2, \dots, \vec{s} _m = \0 \).
    \end{definition}
    \vspace{1pt}
    \begin{definition}
    For subspaces $\vecspace{S}, \vecspace{S}' \subset \vecspace{V}$ we call $\vecspace{S}$ and $\vecspace{S}'$ disjoint if and only if $\vecspace{S} \cap \vecspace{S}' = \{\0\}$
    \end{definition}
    \vspace{1pt}
    \begin{definition}
    For disjoint subspaces $\vecspace{S}, \vecspace{S}' \subset \vecspace{V}$ we define
    \[
        \vecspace{S} \oplus \vecspace{S}' = \{\vec{v}+\vec{u} \ | \ \vec{v}\in\vecspace{S},\vec{u}\in\vecspace{S}\}
    \]
    which is also a subspace of $\vecspace{V}$.
    \end{definition}
